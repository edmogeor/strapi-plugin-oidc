async function logout(ctx: any) {
  const config = strapi.config.get('plugin::strapi-plugin-oidc');
  const logoutUrl = config['OIDC_LOGOUT_URL'];
 
  if (logoutUrl) {
    ctx.redirect(logoutUrl);
  } else {
